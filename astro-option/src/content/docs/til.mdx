import { getCollection } from 'astro:content';

export const tils = (await getCollection('docs')).filter(
  (e) =>
    !['index', 'til'].includes(e.id) &&
    !e.id.startsWith('guides/') &&
    !e.id.startsWith('reference/')
);

export const byCategory = tils.reduce((acc, entry) => {
  const cat = entry.id.split('/')[0];
  (acc[cat] ??= []).push(entry);
  return acc;
}, {});
